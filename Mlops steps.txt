1. Build & Test Locally (Docker)
powershell
# Build image
docker build -f docker/Dockerfile -t ag-news-svm-api:latest .

# Run locally
docker run -p 8000:8000 ag-news-svm-api:latest

# Test
Invoke-WebRequest -Uri "http://localhost:8000/health"
2. Setup Azure Resources
powershell
# Login
az login

# Create Resource Group (Region matters for students!)
az group create --name "ag-news-mlops-rg" --location "spaincentral"

# Create Registry (ACR) - Basic SKU is best for students
$acrName = "agnewsmlops$(Get-Random -Minimum 100000 -Maximum 999999)"
az acr create --resource-group "ag-news-mlops-rg" --name $acrName --sku Basic --admin-enabled true
3. Push to Cloud
powershell
# Login to ACR
az acr login --name $acrName

# Tag Image
docker tag ag-news-svm-api:latest "$acrName.azurecr.io/ag-news-svm-api:latest"

# Push Image
docker push "$acrName.azurecr.io/ag-news-svm-api:latest"
4. Deploy Live API (ACI)
powershell
# Get Password
$acrPassword = az acr credential show --name $acrName --query "passwords[0].value" -o tsv

# Deploy
az container create `
  --resource-group "ag-news-mlops-rg" `
  --name "ag-news-api-container" `
  --image "$acrName.azurecr.io/ag-news-svm-api:latest" `
  --registry-login-server "$acrName.azurecr.io" `
  --registry-username $acrName `
  --registry-password $acrPassword `
  --dns-name-label "agnews-api-$(Get-Random)" `
  --ports 8000 `
  --location "spaincentral" `
  --cpu 1 --memory 1.5 `
  --os-type Linux
